                                design = design,
                                block = metadata$Individual
                                )

voom_trend <- voom(dge,
                   design,
                   block = metadata$Individual, 
                   correlation = cor_fit$consensus.correlation,
                   plot = TRUE)

             correlation = cor_fit$consensus.correlation)

generate_volcano_plot <- function(toptable, comparison_name) {

  volcano_colors <- toptable %>%
    rownames_to_column("Genes") %>%
    left_join(counts %>% 
                select (Ensemble, Symbol),
              by = c("Genes" = "Ensemble")) %>% 
    mutate(
    id_label = ifelse(
      is.na(Symbol),
      Genes,
      Symbol
    )
  ) %>% 
    mutate(
      neglog10AdjP = -log10(pmax(adj.P.Val, 1e-300)),
      color_key = case_when(
        adj.P.Val < 0.05 & logFC > 0  ~ "Up",
        adj.P.Val < 0.05 & logFC < 0 ~ "Down",
        TRUE ~ "NS"
      )
    )

  top_10_genes <- volcano_colors %>%
  arrange(adj.P.Val) %>%
  slice_head(n = 10)
  
  # gene_of_interest <- volcano_colors %>% 
  #   dplyr::filter(id_label == "MCP1")

  p <- ggplot(
    volcano_colors,
    aes(x = logFC, y = neglog10AdjP)
    ) +
    geom_point(
      aes(color = color_key),
      size = 2,
      alpha = 0.6
    ) +
    
   coord_cartesian(
      xlim = c(-12, 12),
      ylim = c(0, 18)
    ) +
    
    # geom_vline(
    # xintercept = c(-1,1),
    # linetype = "dashed"
    # ) +
    # geom_hline(
    # yintercept = -log10(0.05),
    # linetype = "dashed"
    # ) +
    
    geom_text_repel(
      data = top_10_genes,
      #data = gene_of_interest,
      aes(label = id_label),
      size = 3,
      max.overlaps = Inf,
      box.padding = 0.5
    ) +
    
    scale_color_manual(
      values = c(
        "Up" = "red",
        "Down" = "blue",
        "NS" = "grey50"
      )
    ) +
    labs(
      title = comparison_name,
      x = "log2FC",
      y = "-log10(Adj.P-value)"
    ) +
    theme_classic() +
    
    theme(
      plot.title = element_text(hjust = 0.6),
      text = element_text(size = 12)
    ) 
  
    list(
      plot = p,
      top10 = top_10_genes
      # ,
      # interesting_gene = gene_of_interest
      )
}
