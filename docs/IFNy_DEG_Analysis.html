metadata2 <- metadata %>%
  filter(!Condition %in% c("C_IFNy_C2R", "C_VEH_C2R"))

filtered_counts2 <- filtered_raw_counts %>%
  dplyr::select(-c(C_IFNy_C2R, C_VEH_C2R))

# Create DGEList
dge <- DGEList(
  counts = filtered_counts2
)
# TMM normalization
dge <- calcNormFactors(
  dge,
  method = "TMM"
)
# Ensuring my factors are set correctly
metadata2$Species <- relevel(
  factor(metadata2$Species),
  ref = "C")
metadata2$Treatment <- relevel(
  factor(metadata2$Treatment),
  ref = "VEH")
# The code below includes C2R replicate is being considered for normalization, but we are xcluding it, hence why this code is cut out
# metadata$Cond2 <- metadata$Cond
# metadata$Cond2 <- gsub("_R$", "", metadata$Cond2)

# in the 'design' object, we would include '+ Cond2' if we are considering C2R replicate in the following calculations
design <- model.matrix(~ Species * Treatment,
  data = metadata2
)
#colnames(design) <- gsub("^Cond2", "", colnames(design))
colnames(design) <- c(
  "Intercept",
  "SpeciesH",
  "TreatmentIFNy",
  "SpeciesH_TreatmentIFNy"
)

cor_fit <- duplicateCorrelation(object = dge$counts,
                                design = design,
                                block = metadata2$Individual
                                )

voom_trend <- voom(dge,
                   design,
                   block = metadata2$Individual, 
                   correlation = cor_fit$consensus.correlation,
                   plot = FALSE)
# Fit linear model
fit <- lmFit(
  voom_trend,
  design,
  block = metadata2$Individual,
  correlation = cor_fit$consensus.correlation
)
# Defining contrasts
contrast_matrix <- makeContrasts(
  Human_vs_Chimp_VEH = SpeciesH,
  Human_vs_Chimp_IFNy = SpeciesH + SpeciesH_TreatmentIFNy,
  Human_IFNy_vs_VEH = TreatmentIFNy + SpeciesH_TreatmentIFNy,
  Chimp_IFNy_vs_VEH = TreatmentIFNy,
  Interaction = SpeciesH_TreatmentIFNy,
  levels = design)
# Applying contrasts across samples
fit2 <- contrasts.fit(fit, contrast_matrix)
efit2 <- eBayes(fit2)
signif_gene_results = decideTests(efit2, adjust.method = "BH", p.value = 0.05)

generate_volcano_plot <- function(toptable, comparison_name) {

  volcano_colors <- toptable %>%
    rownames_to_column("Genes") %>%
    left_join(counts %>% 
                select(Ensemble, Symbol),
              by = c("Genes" = "Ensemble")) %>% 
    mutate(
    id_label = ifelse(
      is.na(Symbol),
      Genes,
      Symbol
    )
  ) %>% 
    mutate(
      neglog10AdjP = -log10(pmax(adj.P.Val, 1e-300)),
      color_key = case_when(
        adj.P.Val < 0.05 & logFC > 0  ~ "Up",
        adj.P.Val < 0.05 & logFC < 0 ~ "Down",
        TRUE ~ "NS"
      )
    )

  top_genes <- volcano_colors %>%
  arrange(adj.P.Val) %>%
  slice_head(n = 10)
  
  # genes_of_interest <- volcano_colors %>%
  #   dplyr::filter(id_label %in% c("ASPHD2", "C1R", "PARP12", "GBP2", "TNFSF13B", "TUBB", "GRAMD2A", "FAM107A", "ZCWPW2", "TRIB1", "C1S", "CD47", "SFT2D2", "CIBAR1", "IFITM2", "TESK2", "STAMBPL", "NUP58", "BPGM", "CIMAP3", "CD9", "CFH", "SLC35A4", "ENSG00000284048", "UST", "A2M", "CACHD1", "VPS13C", "FBXO31", "NORAD"))

  p <- ggplot(
    volcano_colors,
    aes(x = logFC, y = neglog10AdjP)
    ) +
    geom_point(
      aes(color = color_key),
      size = 2,
      alpha = 0.6
    ) +
    
   # coord_cartesian(
   #    xlim = c(-12, 12),
   #    ylim = c(0, 20)
   #  ) +
    
    # geom_vline(
    # xintercept = c(-1,1),
    # linetype = "dashed"
    # ) +
    # geom_hline(
    # yintercept = -log10(0.05),
    # linetype = "dashed"
    # ) +
    
    geom_text_repel(
      data = top_genes,
      #data = genes_of_interest,
      aes(label = id_label),
      size = 3,
      max.overlaps = Inf,
      box.padding = 0.5
    ) +
    
    scale_color_manual(
      values = c(
        "Up" = "red",
        "Down" = "blue",
        "NS" = "grey50"
      )
    ) +
    labs(
      title = comparison_name,
      x = "log2FC",
      y = "-log10(Adj.P-value)"
    ) +
    theme_classic() +
    
    theme(
      plot.title = element_text(hjust = 0.6),
      text = element_text(size = 12)
    ) 
  
    list(
      plot = p,
      top = top_genes
      #interesting_genes = genes_of_interest
      )
}

generate_GO_plots <- function(toptable, comparison_name){

  bg <- toptable %>%
    tibble::rownames_to_column("Ensembl") %>%
    pull(Ensembl)

  sig <- toptable %>%
    tibble::rownames_to_column("Ensembl") %>%
    filter(adj.P.Val < 0.05) %>%
    pull(Ensembl)

  if(length(sig) == 0){
    message("No significant genes for ", comparison_name)
    return(NULL)
  }

  sig_entrez <- bitr(
    sig,
    fromType = "ENSEMBL",
    toType = "ENTREZID",
    OrgDb = org.Hs.eg.db
  )

  bg_entrez <- bitr(
    bg,
    fromType = "ENSEMBL",
    toType = "ENTREZID",
    OrgDb = org.Hs.eg.db
  )

  ego <- enrichGO(
    gene = sig_entrez$ENTREZID,
    universe = bg_entrez$ENTREZID,
    OrgDb = org.Hs.eg.db,
    keyType = "ENTREZID",
    ont = "BP",
    pAdjustMethod = "BH",
    readable = TRUE
  )

  if (is.null(ego) || nrow(as.data.frame(ego)) == 0) {
  message("No enriched GO terms for ", comparison_name)
  return(NULL)
}
  
  p <- barplot(
    ego,
    showCategory = 20,
    font.size = 10,
    title = paste0("GO_BP ", comparison_name)
  )

  return(p)
}
