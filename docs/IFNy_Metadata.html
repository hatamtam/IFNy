  column_to_rownames(var = "Ensemble") %>% 
  dplyr::select(-c(Entrez_ID, Symbol))
count_matrix <- Ensembl_ID_rownames %>%
  mutate(across(everything(), as.numeric)) %>%
  as.matrix() #This function makes sure the counts table, labeled Ensembl_ID_rownames, is converted to a matrix

log2_cpm <- cpm(count_matrix, log = TRUE, prior.count = 1) #This function assigns the log2_cpm function with the calculation to convert counts to log2 cpm

plot_my_genes <- function(counts_df, gene_id){

gene_tbl <- counts_df %>% 
  dplyr::filter(genes == gene_id)

gene_name <- lookup_table %>% 
  dplyr::filter(rownames(.) == gene_id) %>% 
  pull(Symbol)

gene_long <- gene_tbl %>%
   pivot_longer(-genes, names_to = "Samples", values_to = "log2cpm")

gene_plot <- gene_long %>%
  left_join(metadata, by = c("Samples" = "SampleName")) %>%
  mutate(Treatment = factor(Treatment, levels = c("VEH", "IFNy")))

ggplot(gene_plot, aes(x = Treatment, y = log2cpm)) +
  geom_boxplot(outlier.shape = NA) +
  geom_point(aes(color = Individual)) +
  scale_y_continuous(limits = c(-5, 15),breaks = seq(-5, 15, by = 5)) +
  stat_compare_means(method = "t.test", paired = TRUE, size = 4, label.x = 1.25, label.y = 14) +
  facet_wrap(~Species) +
  labs(title = paste0(gene_name, " Expression Across Conditions"),
       y = "log2 CPM",
       x = "Treatment") +
  theme_bw() +
  theme(plot.title = element_text(hjust = 0.5, size = rel(1.5)),
        axis.text.x = element_text(angle = 45, hjust = 1))
}
